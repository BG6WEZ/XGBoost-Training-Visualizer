#### 2.1.1 新增 Redis 降级适配器

在 `auth.py` 顶部新增：

```python
import os
import json
import logging
from uuid import uuid4
from datetime import datetime, timedelta

logger = logging.getLogger(__name__)

class TokenBlacklist:
    """
    Token 黑名单管理（支持 Redis + 降级）。
    
    工作流：
    1. logout 时，将 JTI 写入 Redis，TTL = token 剩余有效期
    2. 校验 token 时，检查 JTI 是否在黑名单；如果 Redis 不可用，跳过检查
    """
    
    def __init__(self):
        self.redis_client = None
        self._try_init_redis()
    
    def _try_init_redis(self):
        """尝试连接 Redis，失败则记录 warning，不影响启动"""
        try:
            import redis
            redis_url = os.getenv('REDIS_URL', 'redis://localhost:6379/0')
            self.redis_client = redis.from_url(redis_url, decode_responses=True)
            self.redis_client.ping()
            logger.info("✓ Redis 连接成功")
        except Exception as e:
            logger.warning(f"⚠ Redis 不可用，将降级为无黑名单检查: {e}")
            self.redis_client = None
    
    def revoke_token(self, jti: str, exp: int) -> bool:
        """
        将 JTI 写入黑名单，TTL = token 剩余有效期。
        
        Args:
            jti: JWT ID
            exp: Unix timestamp（token 过期时间）
        
        Returns:
            True = 成功写入（或 Redis 不可用）
            False = 黑名单写入失败但需要报错
        """
        if not self.redis_client:
            return True  # 降级：无法写黑名单，也不报错
        
        try:
            ttl_seconds = max(1, exp - int(datetime.utcnow().timestamp()))
            key = f"blacklist:{jti}"
            self.redis_client.setex(key, ttl_seconds, "revoked")
            logger.debug(f"✓ Token {jti} 已加入黑名单（TTL: {ttl_seconds}s）")
            return True
        except Exception as e:
            logger.error(f"✗ 黑名单写入失败: {e}")
            return False
    
    def is_token_revoked(self, jti: str) -> bool:
        """
        检查 JTI 是否在黑名单中。
        
        Args:
            jti: JWT ID
        
        Returns:
            True = 已吊销，False = 有效或 Redis 不可用
        """
        if not self.redis_client:
            return False  # 降级：无法检查黑名单，认为有效
        
        try:
            key = f"blacklist:{jti}"
            exists = self.redis_client.exists(key)
            if exists:
                logger.debug(f"✗ Token {jti} 已被吊销")
            return bool(exists)
        except Exception as e:
            logger.warning(f"⚠ 黑名单检查失败，降级为允许: {e}")
            return False  # 降级：检查失败也不拒绝


# 全局黑名单实例
token_blacklist = TokenBlacklist()
```

#### 2.1.2 修改 `create_access_token()` 增加 JTI

找到 `create_access_token()` 函数，修改为：

```python
def create_access_token(data: dict, expires_delta: Optional[timedelta] = None) -> str:
    """
    创建 JWT token，自动注入 JTI claim。
    """
    to_encode = data.copy()
    
    # 自动注入 JTI（JWT ID，用于黑名单检查）
    to_encode["jti"] = str(uuid4())
    
    if expires_delta:
        expire = datetime.utcnow() + expires_delta
    else:
        expire = datetime.utcnow() + timedelta(minutes=15)
    
    to_encode["exp"] = expire
    
    encoded_jwt = jwt.encode(to_encode, settings.JWT_SECRET, algorithm="HS256")
    return encoded_jwt
```

#### 2.1.3 新增 `revoke_token()` 函数

在认证服务中新增：

```python
def revoke_token(token: str) -> bool:
    """
    吊销一个 token。
    
    Args:
        token: 完整的 JWT token
    
    Returns:
        True = 吊销成功（或无损降级）
    
    Raises:
        ValueError: token 格式错误或 JTI 缺失
    """
    try:
        payload = jwt.decode(token, settings.JWT_SECRET, algorithms=["HS256"])
        jti = payload.get("jti")
        exp = payload.get("exp")
        
        if not jti:
            raise ValueError("Token 中缺少 JTI claim，无法吊销")
        
        # 写入黑名单
        success = token_blacklist.revoke_token(jti, exp)
        if not success:
            logger.error("⚠ 黑名单写入失败，但不中断登出流程")
            # 注意：即使黑名单写入失败，也不在登出时报错
            # 原因：logout 应该总是成功，黑名单仅是加强层
        
        return True
    except jwt.ExpiredSignatureError:
        logger.warning("Token 已过期，无需吊销")
        return True
    except Exception as e:
        logger.error(f"✗ 吊销 token 失败: {e}")
        raise ValueError(f"Unable to revoke token: {str(e)}")
```

#### 2.2.1 新增 logout 端点

在 `auth.py` 路由中新增：

```python
from fastapi import APIRouter, Depends, HTTPException, status
from app.services.auth import revoke_token, get_current_user
from app.models.models import User
import logging

router = APIRouter(prefix="/auth", tags=["auth"])
logger = logging.getLogger(__name__)

@router.post("/logout")
async def logout(current_user: User = Depends(get_current_user), 
                 token: str = Depends(oauth2_scheme)) -> dict:
    """
    登出当前用户。
    
    - 吊销当前 token，使其无法继续使用
    - Redis 不可用时降级为无黑名单检查
    
    Returns:
        { "message": "已登出" }
    
    Raises:
        401: token 无效或已过期
    """
    try:
        revoke_token(token)
        logger.info(f"✓ 用户 {current_user.username} 已登出")
        return {"message": "已登出"}
    except Exception as e:
        logger.error(f"✗ 登出失败: {e}")
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="登出失败"
        )
```

def test_logout_revokes_token(db, client):
    """
    验证：登出后，使用同一 token 请求 /auth/me 返回 401
    
    步骤：
    1. admin 登录获得 token
    2. 调用 /auth/logout

    4. 断言返回 401 Unauthorized

    response = client.post("/api/auth/login", json={
        "username": "admin",
        "password": "admin123"
    })

    token = response.json()["access_token"]
    
    # 2. 登出
    headers = {"Authorization": f"Bearer {token}"}

    assert response.json()["message"] == "已登出"
    
    # 3. 尝试用同一 token 请求 /auth/me
    response = client.get("/api/auth/me", headers=headers)

    assert "已被吊销" in response.json().get("detail", "") or \

## 五、已知约束与风险

1. **Redis 依赖**
   - 若 Redis 配置不正确或宕机，黑名单检查自动降级为无检查
   - 后果：登出后用户仍可用旧 token 访问（安全风险降级但可接受）
   - 建议：生产环境必须配置告警，监控 Redis 可用性

2. **TTL 精度**
   - 黑名单 TTL 计算基于 Unix timestamp，精度为秒
   - 极端场景：token 即将过期时登出可能无法立即删除（Redis 自动过期）

3. **JWT 过期后**
   - Token 过期时自动拒绝（`PyJWTError`），不需要黑名单
   - 黑名单检查仅对有效期内的 token 生效

4. **跨域请求**
   - CORS 需同步配置（Phase-2 Task 2.2 处理）
